### 문제

- [BOJ 14499번 Gold5 주사위 굴리기](https://www.acmicpc.net/problem/14499)

### 풀이 과정

주사위가 이동할 때마다 6면이 바뀌는 규칙을 찾았고 나머지 요구 사항에 따라 구현하였다.

### 코드

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main {
    static class Dice {
        private int top;
        private int bottom;
        private int E;
        private int W;
        private int S;
        private int N;

        public Dice() {
            top = 0;
            bottom = 0;
            E = 0;
            W = 0;
            S = 0;
            N = 0;
        }

        public void rollOver(int order) {
            if(order == 1) {
                rollOverEast();
            } else if(order == 2) {
                rollOverWest();
            } else if(order == 3) {
                rollOverNorth();
            } else {
                rollOverSouth();
            }
        }

        private void rollOverEast() {
            int temp = top;
            top = W;
            W = bottom;
            bottom = E;
            E = temp;
        }

        private void rollOverWest() {
            int temp = top;
            top = E;
            E = bottom;
            bottom = W;
            W = temp;
        }

        private void rollOverNorth() {
            int temp = top;
            top = S;
            S = bottom;
            bottom = N;
            N = temp;
        }

        private void rollOverSouth() {
            int temp = top;
            top = N;
            N = bottom;
            bottom = S;
            S = temp;
        }

        public int getTop() {
            return top;
        }

        public void setBottom(int bottom) {
            this.bottom = bottom;
        }

        public int getBottom() {
            return bottom;
        }
    }
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
        int N = Integer.parseInt(st.nextToken());
        int M = Integer.parseInt(st.nextToken());
        int x = Integer.parseInt(st.nextToken());
        int y = Integer.parseInt(st.nextToken());
        int K = Integer.parseInt(st.nextToken());

        int[][] map = new int[N][M];

        for(int i = 0; i < N; i++) {
            st = new StringTokenizer(br.readLine());
            for(int j = 0; st.hasMoreTokens(); j++) {
                map[i][j] = Integer.parseInt(st.nextToken());
            }
        }
        st = new StringTokenizer(br.readLine());

        int[] orders = new int[K];
        for(int i = 0; st.hasMoreTokens(); i++) {
            orders[i] = Integer.parseInt(st.nextToken());
        }

        solve(map, orders, x, y);

    }

    public static void solve(int[][] map, int[] orders, int x, int y) {
        Dice dice = new Dice();

        StringBuilder sb = new StringBuilder();
        for(int order : orders) {
            if(order == 1 && x < map[0].length-1) {
                x++;
            } else if(order == 2 && x > 0) {
                x--;
            } else if(order == 3 && y > 0) {
                y--;
            } else if(order == 4 && y < map.length-1) {
                y++;
            } else {
                continue;
            }
            dice.rollOver(order);

            if(map[y][x] == 0) {
                map[y][x] = dice.getBottom();
            } else {
                dice.setBottom(map[y][x]);
            }
            sb.append(dice.getTop()).append("\n");
        }

        System.out.println(sb);
    }
}

```
